/**
 * 生成用户授权链接
 *
 * @param string $redirect_uri 微信授权页面配置地址
 * @param string $scope snsapi_base（静默授权）| snsapi_userinfo（非静默授权）
 * @param null $state
 * @return string
 */
public function getAuthURL($redirect_uri, $scope = 'snsapi_userinfo', $state = null)
{
    $params = array();
    $params['appid'] = $this->appid;
    $params['redirect_uri'] = urldecode($redirect_uri);
    $params['response_type'] = 'code';
    $params['scope'] = $scope;
    $params['state'] = $state;
    return "https://open.weixin.qq.com/connect/oauth2/authorize?" . http_build_query($params) . '#wechat_redirect';
}
