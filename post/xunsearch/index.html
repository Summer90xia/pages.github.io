xunsearch结合yii的使用案例

xunSearch安装与启动

  1. 下载安装包
1

wget http://www.xunsearch.com/download/xunsearch-full-latest.tar.bz2
  1. 解压安装包
1

tar -xjf xunsearch-full-latest.tar.bz2
  1. 重命名
1

mv  xunsearch-full-latest.tar.bz2 xunsearch

  1. 进入到 xunsearch 中 执行 ./setup.sh ，选择安装目录为 ~/xunsearch

  2. 启动xunsearch服务

1

bin/xs-ctl.sh -b inet start     // 允许外部任何ip访问
  1. 查看服务是否启动
1

ps -ef | grep xunsearch

// 在app/models中新建XunSearchModel
namespace app\models;

use hightman\xunsearch\ActiveRecord;

class XunSearchModel extends ActiveRecord
{

    public static function projectName()
    {
        return 'search'; // 配置项目使用的config下的 search.ini
    }

}

// 在 service 中实现查询方法

	/**
     * 使用 xunsearch 查询
     * @param $keyword
     * @param $selectType
     * @param int $pageSize
     * @param int $offset
     * @return mixed
     */
    public function getSearchData($keyword, $selectType = 0, $pageSize = 10, $offset = 0)
    {
        $query = XunSearchModel::find()->where($keyword)->andWhere(['status' => 0]);
        if ($selectType) {
            $query->andWhere(['type' => $selectType]);
        }

        $count = $query->count();
        $pages = new Pagination([
            'totalCount' => $count,
            'page'       => Yii::$app->request->get('page', 1) - 1,
            'pageSize'   => $pageSize
        ]);

        $data = $query->offset($offset ?: $pages->offset)
            ->limit($pages->limit)
            ->all();

        // 高亮显示查询结果
        $search = XunSearchModel::getDb()->getSearch();
        foreach ($data as &$val) {
            $val->title = $search->highlight($val->title);
            $val->tags = $search->highlight($val->tags);
            $val->desc = $search->highlight($val->desc);

            $val = $val->toArray();
        }
        
        return $data;
    }
